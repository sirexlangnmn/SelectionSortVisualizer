def selection_sort(arr):
    n = len(arr)
    for i in range(n):
        min_idx = i
        for j in range(i + 1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j
        if min_idx != i:
            arr[i], arr[min_idx] = arr[min_idx], arr[i]
    return arr

scores = [5, 1, 4, 2, 8, 3, 7, 6, 10, 9]
sorted_scores = selection_sort(scores)
print("Sorted scores:", sorted_scores)
